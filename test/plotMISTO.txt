from matplotlib import pyplot as plt

x_values = [(i+1) for i in range(8)]
y_values = [3.442785,1.808318,1.26 ,1.088,0.854,0.755,0.698,0.646]
plt.plot(x_values,y_values, marker='o')
plt.xticks(x_values)
plt.yticks([3.442785,1.808318,1.26 ,1.088,0.646])
plt.grid(True)
plt.xlabel("numero nodi (32 thread per nodo)")
plt.ylabel("tempi")
plt.show()
